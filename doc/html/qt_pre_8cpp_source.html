204 bool qtPre::hasLibQt(CompileTargetBase * container) {
205  bool ok = false;
206  if (!container)
207  return ok;
208 
209  wxArrayString tablibs = container->GetLinkLibs() ;
210  uint16_t nlib = tablibs.GetCount() ;
211  if (nlib > 0 ) {
212  wxString namelib ;
213  uint16_t u=0;
214  int16_t index= -1, pos ;
215  while (u < nlib && !ok ) {
216  // lower, no extension
217  namelib = tablibs.Item(u++).Lower().BeforeFirst('.') ;
218  // no prefix "lib"
219  pos = namelib.Find(_T("lib")) ;
220  if (pos == 0)
221  namelib.Remove(0, 3) ;
222  // begin == "qt"
223  pos = namelib.Find(_T("qt")) ;
224  if (pos != 0)
225  continue ;
226  // compare
227  index = m_TablibQt.Index(namelib);
228  ok = index != -1 ;
229  // first finded
230  if (ok)

232  }
233  }
